    /** 26. Remove Duplicates from Sorted Array 
     * @param nums
     * @return 有序数组中删除重复数字
     */
    public int removeDuplicates(int[] nums) {
    	int count = 0;
    	int len = len = nums.length;
    	for (int i=1; i<len; i++) {
    		if (nums[i] == nums[i-1]) {
    			count++;
    		} else {
    			nums[i-count] = nums[i]; 
    		}
    	}
    	return len-count;
    }
